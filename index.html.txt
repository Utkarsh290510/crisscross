<script>
/* ---------------- GAME LOGIC ---------------- */
let board = ["","","","","","","","",""];
let currentPlayer = "X";
let gameOver = false;

const winPatterns = [
    [0,1,2],[3,4,5],[6,7,8],
    [0,3,6],[1,4,7],[2,5,8],
    [0,4,8],[2,4,6]
];

const cells = document.querySelectorAll(".cell");
const statusText = document.getElementById("status");

cells.forEach(cell => {
    cell.addEventListener("click", () => {
        const i = cell.dataset.index;
        if (board[i] !== "" || gameOver) return;

        board[i] = currentPlayer;
        cell.textContent = currentPlayer;

        checkWin();
        if (!gameOver) switchPlayer();
    });
});

function switchPlayer() {
    currentPlayer = currentPlayer === "X" ? "O" : "X";
    statusText.textContent = `Player ${currentPlayer}'s Turn`;
}

function checkWin() {
    for (let pattern of winPatterns) {
        let [a,b,c] = pattern;
        if (board[a] && board[a] === board[b] && board[b] === board[c]) {
            gameOver = true;
            statusText.textContent = `ðŸŽ‰ Player ${board[a]} Wins!`;
            updateLeaderboard(board[a]);
            return;
        }
    }

    if (!board.includes("")) {
        gameOver = true;
        statusText.textContent = "It's a Draw!";
    }
}

function restart() {
    board = ["","","","","","","","",""];
    currentPlayer = "X";
    gameOver = false;
    statusText.textContent = `Player X's Turn`;
    cells.forEach(c => c.textContent = "");
}

/* ---------------- LEADERBOARD ---------------- */
function loadLeaderboard() {
    const scores = JSON.parse(localStorage.getItem("tttScores")) || {X: 0, O: 0};
    document.getElementById("scores").innerHTML =
        `Player X: <b>${scores.X}</b><br>Player O: <b>${scores.O}</b>`;
}

function updateLeaderboard(winner) {
    const scores = JSON.parse(localStorage.getItem("tttScores")) || {X: 0, O: 0};
    scores[winner]++;
    localStorage.setItem("tttScores", JSON.stringify(scores));
    loadLeaderboard();
}

function resetLeaderboard() {
    localStorage.setItem("tttScores", JSON.stringify({X:0, O:0}));
    loadLeaderboard();
}

loadLeaderboard();
